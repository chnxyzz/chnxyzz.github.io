char* fast_memcpy(char* dest, const char* src, size_t len){
  size_t i;
  // 64-byte block fast copy
  if(len >= 64){
    i = len / 64;
    len &= (64-1);
    while(i-- > 0){
      __asm__ __volatile__ (
      "movdqa (%0), %%xmm0\n"

      "movntps %%xmm0, (%1)\n"
      "movntps %%xmm1, 16(%1)\n"
      "movntps %%xmm2, 32(%1)\n"
      "movntps %%xmm3, 48(%1)\n"
