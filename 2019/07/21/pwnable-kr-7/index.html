from pwn import *

cookie=''
t='1234567890abcdefghijklmnopqrstuvwxyz-_'
def getmessage(message):
    p=remote('pwnable.kr',9006)
    p.recvuntil('ID\n')
    p.sendline(str(message))
    p.recvuntil('PW\n')
    p.sendline()
    p.recvuntil('sending encrypted data (')
    data=p.recvuntil(')')[:-1]
    p.close()
    return data

def getcookie(n):
    global cookie
    s='-'*(62-n%64)
    s1=getmessage(s)
    s1=s1[96:128]
    for i in t:
        s='-'*(63-len(cookie))+cookie+i
        s2=getmessage(s)
        s2=s2[96:128]
        if s1==s2:
            cookie+=i
            print(cookie)
            break
for i in range(1,64):
    getcookie(i)
